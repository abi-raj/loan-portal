create table plan_type(plan_id serial primary key ,name varchar(50),duration int,rate int,penalty int);
create table users(user_id serial primary key,name varchar(30),email varchar(30) unique,age int,phone int,password varchar(30));
create table applications(id serial primary key,plan_id int,user_id int,pan varchar(20),amount int,loan_type varchar(30),remarks varchar(50),foreign key (plan_id) references plan_type(plan_id),foreign key (user_id) references users(user_id) );
